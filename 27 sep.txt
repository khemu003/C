->neon number 
9 ka square :81
8+1 =9 
9 neon number 

#include<stdio.h>
int main(){
    int num=9;
    int square = num*num; //81
    int rem, sum = 0;
    while(square>0){
        rem = square%10;
        sum = sum+rem;
        square = square/10;
    }
    if(num==sum){
        printf("neon");
    }
    else{
        printf("not neon");
    }
    return 0;
}
