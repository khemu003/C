#include<stdio.h>
int main(){
    for(int i=1;i<=5;i++){
        for(int j=1;j<=i;j++){
            printf("*");
        }
        printf("\n");
    }
    
    return 0;
}

#include<stdio.h>
int main(){
    for(int i=1;i<=5;i++){
        for(int j=5;j>=i;j--){
            printf("*");
        }
        printf("\n");
    }
    
    return 0;
}

Fibonacci series 

#include<stdio.h>
int main(){
    int n=10,n1=0,n2=1;
    int next=n1+n2;
    printf("%d %d ",n1,n2);
    for(int i=3;i<=n;i++){
        printf("%d ",next);
        n1 = n2;
        n2 = next;
        next = n1+n2;
    }
    return 0;
}
