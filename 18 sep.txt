-> write a c program input a number and check number is greater than 10 or not ?

#include<stdio.h>
int main(){
    int number;
    printf("Enter a number : ");
    scanf("%d",&number);
    if (number>10){
        printf("number is greater than 10.");
    } else {
        printf("number is not greater than 10.")
    }
    return 0;
}
